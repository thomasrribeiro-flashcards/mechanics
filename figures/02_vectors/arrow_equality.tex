% flashcards-title: Three arrows for comparing vector equality
% flashcards-desc: Arrows P and Q have equal lengths and point up and right from different locations. Arrow R has the same length but points down and left.
\documentclass[border=5mm]{standalone}
\input{figures/tikz-style.tex}
\begin{document}
\begin{tikzpicture}[x=1cm,y=1cm]
  \path[fc canvas] (-.55,-.45) rectangle (10.55,6.65);

  \coordinate (P0) at (.75,5.15);
  \coordinate (P1) at (3.75,6.35);
  \coordinate (Q0) at (.75,1.15);
  \coordinate (Q1) at (3.75,2.35);
  \coordinate (R0) at (9.4,5.15);
  \coordinate (R1) at (6.4,3.95);

  \draw[fc vector] (P0) -- (P1);
  \node[fc endpoint] at (P0) {};
  \node[fc label, below left=2mm and 1mm of P0] {P};

  \draw[fc vector] (Q0) -- (Q1);
  \node[fc endpoint] at (Q0) {};
  \node[fc label, below left=2mm and 1mm of Q0] {Q};

  \draw[fc vector secondary] (R0) -- (R1);
  \node[fc endpoint] at (R0) {};
  \node[fc label, right=2.2mm of R0] {R};
\end{tikzpicture}
\end{document}
